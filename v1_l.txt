import React, { useState, useRef, useEffect } from "react";
import { gsap } from "gsap";
import {
    MdOutlineDriveFileRenameOutline,
    MdDriveFileRenameOutline,
    MdOutlineAlternateEmail,
} from "react-icons/md";
import { FaUserEdit } from "react-icons/fa";
import { PiPassword } from "react-icons/pi";

export default function Register() {
    const [active, setActive] = useState(false); // Gère l'état de la vue active
    const registerRef = useRef(null);
    const loginRef = useRef(null);

    useEffect(() => {
        if (active) {
            // Transition vers Login
            gsap.to(registerRef.current, {
                y: -50,
                autoAlpha: 0,
                duration: 0.5,
                onComplete: () => {
                    gsap.fromTo(
                        loginRef.current,
                        { y: 50, autoAlpha: 0 },
                        { y: 0, autoAlpha: 1, duration: 0.5 }
                    );
                },
            });
        } else {
            // Transition vers Register
            gsap.to(loginRef.current, {
                y: -50,
                autoAlpha: 0,
                duration: 0.5,
                onComplete: () => {
                    gsap.fromTo(
                        registerRef.current,
                        { y: 50, autoAlpha: 0 },
                        { y: 0, autoAlpha: 1, duration: 0.5 }
                    );
                },
            });
        }
    }, [active]);

    return (
        <div className="w-full h-screen flex justify-center items-center  " >
            {/* Section Register */}
            <div
                ref={registerRef}
                className="w-full h-screen flex flex-col sm:flex-row sm:mt-auto  mt-96 justify-center items-center "
                style={{ display: active ? "none" : "flex" }}
            >
                <div className="flex bg-amber-600 shadow-2xl h-140 w-130 justify-center items-center rounded-l-sm flex-col">
                    <form className="flex w-140 py-6 justify-center items-center gap-3 flex-col">
                        <h1 className="font-extrabold cursor-pointer pb-3 text-[#333333] text-4xl">
                            Register
                        </h1>
                        <div className="flex bg-amber-50 items-center shadow-xl w-[300px] gap-2 px-2 rounded-sm">
                            <MdOutlineDriveFileRenameOutline className="w-8 h-8" />
                            <input
                                className="py-3 w-full pr-5 outline-none "
                                type="text"
                                placeholder="First Name"
                            />
                        </div>
                        <div className="flex bg-amber-50 items-center shadow-xl w-[300px] gap-2 px-2 rounded-sm">
                            <MdDriveFileRenameOutline className="w-8 h-8" />
                            <input
                                className="py-3 w-full pr-5 outline-none"
                                type="text"
                                placeholder="Last Name"
                            />
                        </div>
                        <div className="flex bg-amber-50 items-center shadow-xl w-[300px] gap-2 px-2 rounded-sm">
                            <FaUserEdit className="w-8 h-8" />
                            <input
                                className="py-3 w-full pr-5 outline-none"
                                type="text"
                                placeholder="User Name"
                            />
                        </div>
                        <div className="flex bg-amber-50 items-center shadow-xl w-[300px] gap-2 px-2 rounded-sm">
                            <MdOutlineAlternateEmail className="w-8 h-8" />
                            <input
                                className="py-3 w-full pr-5 outline-none"
                                type="email"
                                placeholder="Email"
                            />
                        </div>
                        <div className="flex bg-amber-50 items-center shadow-xl w-[300px] gap-2 px-2 rounded-sm">
                            <PiPassword className="w-8 h-8" />
                            <input
                                className="py-3 w-full pr-5 outline-none"
                                type="password"
                                placeholder="Password"
                            />
                        </div>
                        <div className="py-3">
                            <input
                                className="py-2 bg-[#333333] text-white text-xl px-15 shadow-sm rounded-md focus:outline-none hover:bg-[#1f1e1e] focus:ring-2"
                                type="submit"
                                value="Register"
                            />
                        </div>
                    </form>
                </div>
                <div className="flex bg-[#333333] shadow-2xl h-140 sm:w-[20%] w-full justify-center items-center flex-col">
                    <img src="/test.jpg" className="w-full h-[100%] relative" alt="Register" />

                    <div className="absolute text-center sm:top-40 bottom-50 sm:bottom-auto flex justify-center  items-center flex-col ">
                        <button
                            className="text-white text-2xl font-extrabold border-b-3 mb-5"
                            onClick={() => setActive(true)}
                        >
                            Login
                        </button>
                        <h2 className="font-bold text-lg text-white">
                            Do you already <br /> have an account?
                        </h2>
                    </div>
                </div>
            </div>

            {/* Section Login */}
            <div
                ref={loginRef}
                className="w-full h-screen flex justify-center items-center sm:flex-row   flex-col "
                style={{ display: active ? "flex" : "none" }}
            >
                <div className="flex bg-amber-600 shadow-2xl  sm:h-140   sm:mt-0 mt-54 w-130 justify-center items-center flex-col">
                    <form className="flex w-140 justify-center items-center gap-8 flex-col">
                        <h1 className="font-extrabold cursor-pointer sm:mt-auto mt-10 pb-3 text-[#333333] text-4xl">
                            Login
                        </h1>
                        <div className="flex bg-amber-50 items-center shadow-xl w-[300px] gap-2 px-2 rounded-sm">
                            <MdOutlineAlternateEmail className="w-8 h-8" />
                            <input
                                className="py-3 w-full pr-5 outline-none"
                                type="email"
                                placeholder="Email"
                            />
                        </div>
                        <div className="flex bg-amber-50 items-center shadow-xl w-[300px] gap-2 px-2 rounded-sm">
                            <PiPassword className="w-8 h-8" />
                            <input
                                className="py-3 w-full pr-5 outline-none"
                                type="password"
                                placeholder="Password"
                            />
                        </div>
                        <div className="py-3">
                            <input
                                className="py-2 bg-[#333333] text-white text-xl px-15 shadow-sm rounded-md focus:outline-none hover:bg-[#1f1e1e] focus:ring-2"
                                type="submit"
                                value="Login"
                            />
                        </div>
                    </form>
                </div>
                <div className="flex bg-[#333333] shadow-2xl h-140  sm:w-[20%] w-full justify-center items-center flex-col">
                    <img src="/test.jpg" className="w-full h-[100%] relative" alt="Login" />
                    <div className="absolute text-center flex flex-col   justify-center items-center mb-64">
                        <button
                            className="text-white text-2xl font-extrabold border-b-3 mb-5"
                            onClick={() => setActive(false)}
                        >
                            Register
                        </button>
                        <h2 className="font-bold text-lg text-white">
                            Don't have <br /> an account already?
                        </h2>
                    </div>
                </div>
            </div>
        </div>
    );
} fixe responsive
